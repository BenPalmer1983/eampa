! Set Variables
    dataSetSize = size(points,1)
! Handle optional arguments  
    inputSetStart = 1
  inputSetLength = dataSetSize
  If(Present(inputSetStartIn))Then
    inputSetStart = inputSetStartIn
  End If
  If(Present(inputSetLengthIn))Then
    inputSetLength = inputSetLengthIn
  End If
  inputSetEnd = inputSetStart+inputSetLength-1
! Check data set size
  If(subsetSize.lt.2)Then
    subsetSize = 2
  End If
  If(subsetSize.gt.dataSetSize)Then
    subsetSize = dataSetSize
  End If

! Reduce set of data points  
  If(subsetSize.eq.dataSetSize)Then
    pointsInterp = points
  Else If(subsetSize.lt.dataSetSize)Then
!reduce set of data points
      xLower = points(inputSetStart,1)
    xUpper = points(inputSetEnd,1)
    posOffset = 0
    If(x.lt.xLower)Then
      xPos = 1
    posOffset = 0
    Elseif(x.gt.dataSetSize)Then
      xPos = dataSetSize
    posOffset = subsetSize
    Else
  !Estimate position
    xPos = Floor(((x - xLower) / (xUpper - xLower)) * 1.0D0 * dataSetSize) + inputSetStart
    xLower = points(xPos,1)
    xUpper = points(xPos+1,1)
    If(x.ge.xLower.and.x.le.xUpper)Then
!Position found  
    Else
!Find position  
      Do i=1,dataSetSize
        xLower = points(xPos+i,1)
        xUpper = points(xPos+i+1,1)
        If((xPos+i+1).eq.inputSetEnd.or.(x.ge.xLower.and.x.le.xUpper))Then
        xPos = xPos + i
        Exit
      End If
        xLower = points(xPos-i,1)
        xUpper = points(xPos-i+1,1)
        If((xPos-i).eq.inputSetStart.or.(x.ge.xLower.and.x.le.xUpper))Then
        xPos = xPos-i
        Exit
      End If      
      End Do
!Set offset
          posOffset = -1*floor(subsetSize/2.0D0)
    End If
    End If
!adjust position with offset
    If((xPos+posOffset).lt.inputSetStart)Then
    posOffset = 0  
    End If
    If((xPos-posOffset).gt.inputSetEnd)Then
      posOffset = -1*subsetSize  
    End If
    If((xPos+subsetSize).gt.inputSetEnd)Then
      posOffset = -1*subsetSize  
    End If
    Do i=1,subsetSize
    pointsInterp(i,1) = points(xPos+posOffset+i-1,1)
    pointsInterp(i,2) = points(xPos+posOffset+i-1,2)
    End Do
    dataSetSize = subsetSize    
  End If
  
  Do i=1,dataSetSize
    print *,pointsInterp(i,1),pointsInterp(i,2)
  End Do
